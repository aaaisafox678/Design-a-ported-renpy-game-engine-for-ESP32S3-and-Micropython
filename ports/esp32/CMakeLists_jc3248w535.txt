cmake_minimum_required(VERSION 3.16)

set(JC3248W535CN_DIR "${CMAKE_CURRENT_LIST_DIR}")

# Board-specific configuration
set(MICROPY_BOARD JC3248W535CN)
set(MICROPY_BOARD_DIR "${MICROPY_BINDIR}/boards/${MICROPY_BOARD}")

# ESP32-S3 specific settings
set(MICROPY_FROZEN_MANIFEST "${MICROPY_SOURCE_DIR}/ports/esp32/manifest_renpy.py")

# Chip configuration
set(MICROPY_CHIP_VARIANT ESP32S3)
set(MICROPY_PY_NETWORK 1)
set(MICROPY_PY_NETWORK_CYW43 0)

# PSRAM configuration - Octal PSRAM for JC3248W535CN
set(MICROPY_HW_ENABLE_PSRAM 1)

# Extra C definitions for this board
set(MICROPY_CFLAGS_BOARD_EXTRA
    "-DMICROPY_CONFIG_ROM_LEVEL=EXTENSIVE"
    "-DJC3248W535CN"
    "-DMICROPY_HW_LCD_H_RES=320"
    "-DMICROPY_HW_LCD_V_RES=480"
    "-DMICROPY_HW_LCD_QSPI"
    "-DMICROPY_HW_TOUCH_I2C"
    "-DMICROPY_HW_AUDIO_I2S"
)

# LVGL Configuration
set(MICROPY_CFLAGS_BOARD_EXTRA "${MICROPY_CFLAGS_BOARD_EXTRA};-DMICROPY_LVGL_ENABLE=1")

# Add board-specific sources
set(MICROPY_PORT_SRCS
    "${JC3248W535CN_DIR}/board.c"
    "${JC3248W535CN_DIR}/pins_jc3248w535.c"
    "${MICROPY_SOURCE_DIR}/extmod/extmod/lvgl/lvgl_port.c"
    "${MICROPY_SOURCE_DIR}/extmod/extmod/lvgl/lvgl_port_disp.c"
    "${MICROPY_SOURCE_DIR}/extmod/extmod/lvgl/lvgl_port_indev.c"
)

# Link libraries
set(MICROPY_EXTRA_LIBS "")

# Include board directory
list(APPEND MICROPY_EXTRA_SOURCE_DIRS "${JC3248W535CN_DIR}")

# Micro-RenPy module source files
set(RENPY_MODULE_SRCS
    "${MICROPY_SOURCE_DIR}/ports/esp32/mp_renpy.c"
    "${MICROPY_SOURCE_DIR}/ports/esp32/mp_renpy_display.c"
    "${MICROPY_SOURCE_DIR}/ports/esp32/mp_renpy_input.c"
    "${MICROPY_SOURCE_DIR}/ports/esp32/mp_renpy_audio.c"
    "${MICROPY_SOURCE_DIR}/ports/esp32/mp_renpy_script.c"
)

list(APPEND MICROPY_EXTRA_SOURCE_FILES "${RENPY_MODULE_SRCS}")

# Set the partition table
set(MICROPY_PARTITION_TINYFS_2MB "tinyfs, data, fat, 0x80000,," CACHE STRING "Partition table for tinyfs")

# Flash settings
set(MICROPY_BOARD_ESP_SDK_LIB_PATH "${IDF_PATH}/lib")

# PSRAM speed configuration for MJPEG playback
# Must be set to 120M for optimal performance
# This matches the requirement from the Arduino DEMO
set(MICROPY_PSRAM_CLOCK_DIV 5)  # 240MHz / 2 = 120MHz PSRAM clock

# End of file
